Plot all maturity curves on one graph

plt.figure(figsize=(12,6))
for col in yld_rts.columns:
    plt.plot(yld_rts.index, yld_rts[col], label=col, linewidth=1)
plt.legend(loc='best', fontsize='small', ncol=2)
plt.xlabel('Date')
